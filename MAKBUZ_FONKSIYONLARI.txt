  const handleDownloadReceipt = async (due: any) => {
    try {
      // İzin kontrolü
      const { status } = await MediaLibrary.requestPermissionsAsync();
      
      if (status !== 'granted') {
        Alert.alert('İzin Gerekli', 'Makbuzu kaydetmek için medya kütüphanesine erişim izni gerekiyor.');
        return;
      }

      Alert.alert(
        'Makbuz İndir',
        'Ödeme makbuzunuzu PDF olarak kaydetmek istiyor musunuz?',
        [
          { text: 'İptal', style: 'cancel' },
          {
            text: 'Kaydet',
            onPress: async () => {
              try {
                // Generate PDF
                const html = generateReceiptHTML(due);
                const { uri } = await Print.printToFileAsync({ html });
                
                // Dosya adı oluştur
                const fileName = `Makbuz_${due.period}_${Date.now()}.pdf`;
                
                // Geçici dosyayı kalıcı konuma taşı
                const fileUri = FileSystem.documentDirectory + fileName;
                await FileSystem.copyAsync({
                  from: uri,
                  to: fileUri,
                });
                
                // MediaLibrary'ye kaydet (Downloads klasörüne)
                const asset = await MediaLibrary.createAssetAsync(fileUri);
                await MediaLibrary.createAlbumAsync('Download', asset, false);
                
                Alert.alert('Başarılı', `Makbuz indirildi:\n${fileName}\n\nİndirilenler klasöründen erişebilirsiniz.`);
              } catch (error) {
                console.error('PDF save error:', error);
                Alert.alert('Hata', 'Makbuz kaydedilirken bir hata oluştu');
              }
            }
          }
        ]
      );
    } catch (error) {
      console.error('Download receipt error:', error);
      Alert.alert('Hata', 'Makbuz kaydedilemedi');
    }
  };
